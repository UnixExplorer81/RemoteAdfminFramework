function New-MountPsDriveObject {
    param (
        [Parameter(Mandatory)]
        [string]$UNCPath,

        [Parameter(Mandatory)]
        [pscredential]$Credential,

        [string]$DriveName = $null,
        [switch]$AutoUnmount
    )

    $UNCPath = $UNCPath.TrimEnd('\')
    $server  = ($UNCPath -replace '^\\\\([^\\]+)\\.*', '$1')
    $share   = ($UNCPath -replace '^\\\\[^\\]+\\([^\\]+)\\?.*', '$1')
    $drive   = if ($DriveName) { $DriveName } else { "$server-$share" }

    $psDriveName = "$drive:"
    $mounted = $false

    # Check if drive exists
    $existing = Get-PSDrive -Name $drive -ErrorAction SilentlyContinue
    if ($existing -and ($existing.Root -eq $UNCPath)) {
        Write-Verbose "Drive '$drive' already exists and matches path."
        $mounted = $true
    } else {
        try {
            New-PSDrive -Name $drive -PSProvider FileSystem -Root $UNCPath -Credential $Credential -Persist:$false -Scope Global -ErrorAction Stop | Out-Null
            $mounted = $true
        } catch {
            throw "❌ Failed to mount drive '$drive' to '$UNCPath': $_"
        }
    }

    if ($AutoUnmount) {
        Register-EngineEvent PowerShell.Exiting -Action {
            try {
                Remove-PSDrive -Name $using:drive -Force -ErrorAction SilentlyContinue
            } catch {}
        } | Out-Null
    }

    return @{
        DriveName   = $drive
        UNCPath     = $UNCPath
        Server      = $server
        Share       = $share
        Credential  = $Credential
        Mounted     = $mounted
        PSDriveName = $psDriveName
        AutoUnmount = $AutoUnmount.IsPresent
    }
}

function Get-PathFromMount {
    param (
        [Parameter(Mandatory)]
        [hashtable]$DriveObject,

        [string]$Path = ""
    )

    if ($Path -like "\\\\*") {
        $relativePath = $Path.Substring($DriveObject.UNCPath.Length).TrimStart('\')
    } else {
        $relativePath = $Path
    }

    return Join-Path -Path $DriveObject.PSDriveName -ChildPath $relativePath
}

function Unmount-PsDriveObject {
    param (
        [Parameter(Mandatory)]
        [hashtable]$DriveObject
    )

    if ($DriveObject.Mounted -and (Get-PSDrive -Name $DriveObject.DriveName -ErrorAction SilentlyContinue)) {
        try {
            Remove-PSDrive -Name $DriveObject.DriveName -Force -ErrorAction Stop
            $DriveObject.Mounted = $false
        } catch {
            Write-Warning "⚠️ Failed to remove PSDrive '$($DriveObject.DriveName)': $_"
        }
    }
}
